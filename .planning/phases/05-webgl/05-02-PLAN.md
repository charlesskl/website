**Create `assets/animation/card-effects.js`** — Card clip-path scroll reveals and 3D mouse-tilt hover.

IIFE pattern with 'use strict'. Register via `window.RR.register('cardEffects', {init, kill, refresh})`.

The module must:

1. **Page gate**: Only meaningful on pages with `.cap-card` elements. If `document.querySelectorAll('.cap-card').length === 0`, return from init(). No page name check needed — the selector check is sufficient and more robust.

2. **CDN graceful degradation**: If GSAP or ScrollTrigger missing, return. Cards remain visible with existing CSS hover effects.

3. **Reduced motion gate**: If `window.RR.state.hasReducedMotion`, ensure all cards are immediately visible (no clip-path hiding) and skip tilt. Do NOT hide cards and leave them invisible.

4. **CARD SCROLL REVEAL** (CARD-01):

   The existing scroll-reveals.js already handles `.cap-card-img` clip-path wipe. card-effects.js should handle the WHOLE CARD entry animation — staggering the cards as a group:

   ```javascript
   function initCardReveals() {
     var cards = document.querySelectorAll('.cap-card');
     if (!cards.length) return;

     var grid = document.querySelector('.caps-grid');
     if (!grid) return;

     // Set initial state: cards start invisible with clip-path
     cards.forEach(function(card) {
       gsap.set(card, {
         clipPath: 'inset(100% 0 0 0)',  // clipped from bottom — invisible
         opacity: 0
       });
     });

     // Staggered reveal timeline
     var tl = gsap.timeline({
       scrollTrigger: {
         trigger: grid,
         start: 'top 80%',
         once: true
       }
     });

     tl.to(cards, {
       clipPath: 'inset(0% 0 0 0)',  // reveal from top
       opacity: 1,
       duration: 0.8,
       ease: 'power3.inOut',
       stagger: {
         each: 0.12,
         from: 'start'  // left-to-right stagger
       }
     });

     trackedTimelines.push(tl);
     if (tl.scrollTrigger) trackedTriggers.push(tl.scrollTrigger);
   }
   ```

   IMPORTANT: This clip-path reveal is for the ENTIRE card element. The existing scroll-reveals.js `.cap-card-img` wipe will ALSO run (it's a separate nested animation within each card). The card-level clip-path creates the entry, then the image wipe adds detail once the card is visible. If this creates visual conflict, set `initCardReveals()` to use the `.caps-grid` scroll trigger at `top 85%` (slightly later than scroll-reveals.js which uses `top 80%` on individual cards), so the card reveals first, then images wipe in.

   Alternative if there IS conflict: card-effects.js should TAKE OVER the cap-card-img reveal too, consolidating it from scroll-reveals.js. To avoid double-animation, card-effects.js should check if scroll-reveals.js has already set clipPath on `.cap-card-img` and skip those.

5. **3D TILT HOVER** (CARD-02, CARD-03):

   Desktop-only 3D tilt that follows mouse position:

   ```javascript
   function initCardTilt() {
     // Touch device gate (CARD-02)
     var isTouch = (window.RR && window.RR.breakpoints && window.RR.breakpoints.isCoarsePointer) ||
                   window.matchMedia('(pointer: coarse)').matches;
     if (isTouch) return;

     var cards = document.querySelectorAll('.cap-card');

     cards.forEach(function(card) {
       // Add perspective to the card's parent for 3D context
       card.style.transformStyle = 'preserve-3d';

       var mouseEnter = function() {
         // Add will-change during interaction (CARD-03, PERF-03)
         card.style.willChange = 'transform';
       };

       var mouseMove = function(e) {
         var rect = card.getBoundingClientRect();
         // Calculate mouse position relative to card center (-0.5 to 0.5)
         var x = (e.clientX - rect.left) / rect.width - 0.5;
         var y = (e.clientY - rect.top) / rect.height - 0.5;

         // Tilt angles (max ±8 degrees)
         var tiltX = y * -16;  // inverted Y → X rotation
         var tiltY = x * 16;   // X → Y rotation

         // Apply with GSAP for smooth interpolation (CARD-03)
         gsap.to(card, {
           rotateX: tiltX,
           rotateY: tiltY,
           transformPerspective: 800,
           duration: 0.4,
           ease: 'power2.out',
           overwrite: 'auto'  // kill previous tween
         });
       };

       var mouseLeave = function() {
         // Return to flat (CARD-03 — GPU-compositable only)
         gsap.to(card, {
           rotateX: 0,
           rotateY: 0,
           duration: 0.6,
           ease: 'elastic.out(1, 0.5)',
           overwrite: 'auto',
           onComplete: function() {
             // Remove will-change after animation settles (PERF-03)
             card.style.willChange = 'auto';
           }
         });
       };

       card.addEventListener('mouseenter', mouseEnter);
       card.addEventListener('mousemove', mouseMove);
       card.addEventListener('mouseleave', mouseLeave);

       // Track for cleanup
       cleanupFns.push(function() {
         card.removeEventListener('mouseenter', mouseEnter);
         card.removeEventListener('mousemove', mouseMove);
         card.removeEventListener('mouseleave', mouseLeave);
         card.style.willChange = 'auto';
         card.style.transformStyle = '';
         gsap.set(card, { clearProps: 'rotateX,rotateY,transformPerspective' });
       });
     });
   }
   ```

   Key details:
   - `transformPerspective: 800` on each card (applied via GSAP) creates the 3D effect without needing a parent perspective (GSAP applies it inline).
   - Max tilt: ±8 degrees (16 * 0.5) — subtle enough to be premium, not gimmicky.
   - `elastic.out` ease on leave creates a satisfying spring-back.
   - `overwrite: 'auto'` prevents tween stacking during rapid mousemove.
   - Only `transform` property is animated (CARD-03) — no filter, no layout properties.

8. **kill()**:
   ```javascript
   function kill() {
     // Run all cleanup functions (removes event listeners, resets styles)
     cleanupFns.forEach(function(fn) { try { fn(); } catch(e) {} });
     cleanupFns = [];

     // Kill ScrollTrigger timelines
     trackedTimelines.forEach(function(tl) { try { tl.kill(); } catch(e) {} });
     trackedTimelines = [];

     trackedTriggers.forEach(function(st) { try { st.kill(); } catch(e) {} });
     trackedTriggers = [];

     // Reset clip-path on all cards
     var cards = document.querySelectorAll('.cap-card');
     cards.forEach(function(card) {
       gsap.set(card, { clearProps: 'clipPath,opacity,rotateX,rotateY,transformPerspective' });
     });
   }
   ```
